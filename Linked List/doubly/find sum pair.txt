import java.util.*;
class Node
{
    int data;
    Node next;
    Node prev;
    
    public Node(int data)
    {
        this.data=data;
        this.next=null;
        this.prev=null;
    }
}
class DLL
{
    public void print(Node head)
    {
        Node curr=head;
        while(curr!=null)
        {
            System.out.print(curr.data+"->");
            curr=curr.next;
        }
        System.out.print("null");
        System.out.println();
    }
    
    public void findPair(Node head,Node tail,int x)
    {
        Node prev=head;
        Node curr=tail;
        int count=0;
        while(prev!=curr && curr.next!=prev)
        {
            if(prev.data+curr.data < x)
            {
                prev=prev.next;
            }
            else if(prev.data+curr.data > x)
            {
                curr=curr.prev;
            }
            else
            {
                count=1;
                System.out.println("pair is: "+"("+prev.data+","+curr.data+")");
                prev=prev.next;
                curr=curr.prev;
            }
        }
        if(count==0)
        {
            System.out.println("NO PAIR FOUND");
        }
    }
}

public class Main
{
	public static void main(String[] args)
	{
	    DLL ll=new DLL();
	    Scanner s=new Scanner(System.in);
	    int n=s.nextInt();
	    Node head=new Node(s.nextInt());
	    Node tail=head;
	    while(n!=1)
	    {
	        tail.next=new Node(s.nextInt());
	        tail.next.prev=tail;
	        tail=tail.next;
	        n--;
	    }
	    ll.findPair(head,tail,7);
	    ll.print(head);
		
	}
}
