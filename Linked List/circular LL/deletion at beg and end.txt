import java.util.*;
class Node
{
    int data;
    Node prev;
    Node next;
    public Node(int data)
    {
        this.data=data;
        this.next=null;
    }
}
class CLL
{
    public void print(Node head)
    {
        Node curr=head;
        while(curr.next!=head)
        {
            System.out.print(curr.data+"->");
            curr=curr.next;
        }
        System.out.print(curr.data);
        System.out.println();
    }
    
    // delete at beginnning
    public Node deleteAtBeg(Node head)
    {
        if(head==null)
        {
            return head;
        }
        Node tail=head;
        while(tail.next!=head)
        {
            tail=tail.next;
        }
        tail.next=head.next;
        head=head.next;
        return head;
    }
    
    // delete at end
    public Node deleteAtEnd(Node head)
    {
        Node tail=head;
        while(tail.next.next!=head)
        {
            tail=tail.next;
        }
        tail.next=head;
        return head;
    }
}
public class Main
{
	public static void main(String[] args) 
	{
	    Scanner s=new Scanner(System.in);
	    int n=s.nextInt();
	    CLL ll=new CLL();
	    Node head=new Node(s.nextInt());
	    Node tail=head;
	    while(n!=1)
	    {
	        tail.next=new Node(s.nextInt());
	        tail=tail.next;
	        n--;
	    }
	    tail.next=head;
	    ll.print(head);
	    head=ll.deleteAtBeg(head);
	    ll.print(head);
	    
	    head=ll.deleteAtEnd(head);
	    ll.print(head);
	    
	}
}
